258 def on(self, message, namespace=None): 

259 """Decorator to register a SocketIO event handler. 

260 

261 This decorator must be applied to SocketIO event handlers. Example:: 

262 

263 @socketio.on('my event', namespace='/chat') 

264 def handle_my_custom_event(json): 

265 print('received json: ' + str(json)) 

266 

267 :param message: The name of the event. This is normally a user defined 

268 string, but a few event names are already defined. Use 

269 ``'message'`` to define a handler that takes a string 

270 payload, ``'json'`` to define a handler that takes a 

271 JSON blob payload, ``'connect'`` or ``'disconnect'`` 

272 to create handlers for connection and disconnection 

273 events. 

274 :param namespace: The namespace on which the handler is to be 

275 registered. Defaults to the global namespace. 

276 """ 

277 namespace = namespace or '/' 

278 

279 def decorator(handler): 

280 @wraps(handler) 

281 def _handler(sid, *args): 

282 return self._handle_event(handler, message, namespace, sid, 

283 *args) 

284 

285 if self.server: 

286 self.server.on(message, _handler, namespace=namespace) 

287 else: 

288 self.handlers.append((message, _handler, namespace)) 

289 return handler 

290 return decorator

366 def emit(self, event, *args, **kwargs): 

367 """Emit a server generated SocketIO event. 

368 

369 This function emits a SocketIO event to one or more connected clients. 

370 A JSON blob can be attached to the event as payload. This function can 

371 be used outside of a SocketIO event context, so it is appropriate to 

372 use when the server is the originator of an event, outside of any 

373 client context, such as in a regular HTTP request handler or a 

374 background task. Example:: 

375 

376 @app.route('/ping') 

377 def ping(): 

378 socketio.emit('ping event', {'data': 42}, namespace='/chat') 

379 

380 :param event: The name of the user event to emit. 

381 :param args: A dictionary with the JSON data to send as payload. 

382 :param namespace: The namespace under which the message is to be sent. 

383 Defaults to the global namespace. 

384 :param room: Send the message to all the users in the given room. If 

385 this parameter is not included, the event is sent to 

386 all connected users. 

387 :param include_self: ``True`` to include the sender when broadcasting 

388 or addressing a room, or ``False`` to send to 

389 everyone but the sender. 

390 :param skip_sid: The session id of a client to ignore when broadcasting 

391 or addressing a room. This is typically set to the 

392 originator of the message, so that everyone except 

393 that client receive the message. To skip multiple sids 

394 pass a list. 

395 :param callback: If given, this function will be called to acknowledge 

396 that the client has received the message. The 

397 arguments that will be passed to the function are 

398 those provided by the client. Callback functions can 

399 only be used when addressing an individual client. 

400 """ 

401 namespace = kwargs.pop('namespace', '/') 

402 room = kwargs.pop('room', None) 

403 include_self = kwargs.pop('include_self', True) 

404 skip_sid = kwargs.pop('skip_sid', None) 

405 if not include_self and not skip_sid: 

406 skip_sid = flask.request.sid 

407 callback = kwargs.pop('callback', None) 

408 if callback: 

409 # wrap the callback so that it sets app app and request contexts 

410 sid = None 

411 if has_request_context: 

412 sid = getattr(flask.request, 'sid', None) 

413 original_callback = callback 

414 

415 def _callback_wrapper(*args): 

416 return self._handle_event(original_callback, None, namespace, 

417 sid, *args) 

418 

419 if sid: 

420 # the callback wrapper above will install a request context 

421 # before invoking the original callback 

422 # we only use it if the emit was issued from a Socket.IO 

423 # populated request context (i.e. request.sid is defined) 

424 callback = _callback_wrapper 

425 self.server.emit(event, *args, namespace=namespace, room=room, 

426 skip_sid=skip_sid, callback=callback, **kwargs)

685 def _handle_event(self, handler, message, namespace, sid, *args): 

686 if sid not in self.server.environ: 

687 # we don't have record of this client, ignore this event 

688 return '', 400 

689 app = self.server.environ[sid]['flask.app'] 

690 with app.request_context(self.server.environ[sid]): 

691 if self.manage_session: 

692 # manage a separate session for this client's Socket.IO events 

693 # created as a copy of the regular user session 

694 if 'saved_session' not in self.server.environ[sid]: 

695 self.server.environ[sid]['saved_session'] = \ 

696 _ManagedSession(flask.session) 

697 session_obj = self.server.environ[sid]['saved_session'] 

698 else: 

699 # let Flask handle the user session 

700 # for cookie based sessions, this effectively freezes the 

701 # session to its state at connection time 

702 # for server-side sessions, this allows HTTP and Socket.IO to 

703 # share the session, with both having read/write access to it 

704 session_obj = flask.session._get_current_object() 

705 _request_ctx_stack.top.session = session_obj 

706 flask.request.sid = sid 

707 flask.request.namespace = namespace 

708 flask.request.event = {'message': message, 'args': args} 

709 try: 

710 if message == 'connect': 

711 ret = handler() 

712 else: 

713 ret = handler(*args) 

714 except: 

715 err_handler = self.exception_handlers.get( 

716 namespace, self.default_exception_handler) 

717 if err_handler is None: 

718 raise 

719 type, value, traceback = sys.exc_info() 

720 return err_handler(value) 

721 if not self.manage_session: 

722 # when Flask is managing the user session, it needs to save it 

723 if not hasattr(session_obj, 'modified') or session_obj.modified: 

724 resp = app.response_class() 

725 app.session_interface.save_session(app, session_obj, resp) 

726 return ret

729def emit(event, *args, **kwargs): 

730 """Emit a SocketIO event. 

731 

732 This function emits a SocketIO event to one or more connected clients. A 

733 JSON blob can be attached to the event as payload. This is a function that 

734 can only be called from a SocketIO event handler, as in obtains some 

735 information from the current client context. Example:: 

736 

737 @socketio.on('my event') 

738 def handle_my_custom_event(json): 

739 emit('my response', {'data': 42}) 

740 

741 :param event: The name of the user event to emit. 

742 :param args: A dictionary with the JSON data to send as payload. 

743 :param namespace: The namespace under which the message is to be sent. 

744 Defaults to the namespace used by the originating event. 

745 A ``'/'`` can be used to explicitly specify the global 

746 namespace. 

747 :param callback: Callback function to invoke with the client's 

748 acknowledgement. 

749 :param broadcast: ``True`` to send the message to all clients, or ``False`` 

750 to only reply to the sender of the originating event. 

751 :param room: Send the message to all the users in the given room. If this 

752 argument is set, then broadcast is implied to be ``True``. 

753 :param include_self: ``True`` to include the sender when broadcasting or 

754 addressing a room, or ``False`` to send to everyone 

755 but the sender. 

756 :param skip_sid: The session id of a client to ignore when broadcasting 

757 or addressing a room. This is typically set to the 

758 originator of the message, so that everyone except 

759 that client receive the message. To skip multiple sids 

760 pass a list. 

761 :param ignore_queue: Only used when a message queue is configured. If 

762 set to ``True``, the event is emitted to the 

763 clients directly, without going through the queue. 

764 This is more efficient, but only works when a 

765 single server process is used, or when there is a 

766 single addressee. It is recommended to always leave 

767 this parameter with its default value of ``False``. 

768 """ 

769 if 'namespace' in kwargs: 

770 namespace = kwargs['namespace'] 

771 else: 

772 namespace = flask.request.namespace 

773 callback = kwargs.get('callback') 

774 broadcast = kwargs.get('broadcast') 

775 room = kwargs.get('room') 

776 if room is None and not broadcast: 

777 room = flask.request.sid 

778 include_self = kwargs.get('include_self', True) 

779 skip_sid = kwargs.get('skip_sid') 

780 ignore_queue = kwargs.get('ignore_queue', False) 

781 

782 socketio = flask.current_app.extensions['socketio'] 

783 return socketio.emit(event, *args, namespace=namespace, room=room, 

784 include_self=include_self, skip_sid=skip_sid, 

785 callback=callback, ignore_queue=ignore_queue)

788def send(message, **kwargs):

794 can only be called from a SocketIO event handler. 

795 

796 :param message: The message to send, either a string or a JSON blob. 

797 :param json: ``True`` if ``message`` is a JSON blob, ``False`` 

798 otherwise. 

799 :param namespace: The namespace under which the message is to be sent. 

800 Defaults to the namespace used by the originating event. 

801 An empty string can be used to use the global namespace. 

802 :param callback: Callback function to invoke with the client's 

803 acknowledgement. 

804 :param broadcast: ``True`` to send the message to all connected clients, or 

805 ``False`` to only reply to the sender of the originating 

806 event. 

807 :param room: Send the message to all the users in the given room. 

808 :param include_self: ``True`` to include the sender when broadcasting or 

809 addressing a room, or ``False`` to send to everyone 

810 but the sender. 

811 :param skip_sid: The session id of a client to ignore when broadcasting 

812 or addressing a room. This is typically set to the 

813 originator of the message, so that everyone except 

814 that client receive the message. To skip multiple sids 

815 pass a list. 

816 :param ignore_queue: Only used when a message queue is configured. If 

817 set to ``True``, the event is emitted to the 

818 clients directly, without going through the queue. 

819 This is more efficient, but only works when a 

820 single server process is used, or when there is a 

821 single addressee. It is recommended to always leave 

822 this parameter with its default value of ``False``. 

823 """ 

824 json = kwargs.get('json', False) 

825 if 'namespace' in kwargs: 

826 namespace = kwargs['namespace'] 

827 else: 

828 namespace = flask.request.namespace 

829 callback = kwargs.get('callback') 

830 broadcast = kwargs.get('broadcast') 

831 room = kwargs.get('room') 

832 if room is None and not broadcast: 

833 room = flask.request.sid 

834 include_self = kwargs.get('include_self', True) 

835 skip_sid = kwargs.get('skip_sid') 

836 ignore_queue = kwargs.get('ignore_queue', False) 

837 

838 socketio = flask.current_app.extensions['socketio'] 

839 return socketio.send(message, json=json, namespace=namespace, room=room,
