79def set_nonblocking(fd): 

80 """ 

81 Sets the descriptor to be nonblocking. Works on many file-like 

82 objects as well as sockets. Only sockets can be nonblocking on 

83 Windows, however. 

84 """ 

85 try: 

86 setblocking = fd.setblocking 

87 except AttributeError: 

88 # fd has no setblocking() method. It could be that this version of 

89 # Python predates socket.setblocking(). In that case, we can still set 

90 # the flag "by hand" on the underlying OS fileno using the fcntl 

91 # module. 

92 try: 

93 import fcntl 

94 except ImportError: 

95 # Whoops, Windows has no fcntl module. This might not be a socket 

96 # at all, but rather a file-like object with no setblocking() 

97 # method. In particular, on Windows, pipes don't support 

98 # non-blocking I/O and therefore don't have that method. Which 

99 # means fcntl wouldn't help even if we could load it. 

100 raise NotImplementedError("set_nonblocking() on a file object " 

101 "with no setblocking() method " 

102 "(Windows pipes don't support non-blocking I/O)") 

103 # We managed to import fcntl. 

104 fileno = fd.fileno() 

105 orig_flags = fcntl.fcntl(fileno, fcntl.F_GETFL) 

106 new_flags = orig_flags | os.O_NONBLOCK 

107 if new_flags != orig_flags: 

108 fcntl.fcntl(fileno, fcntl.F_SETFL, new_flags) 

109 else: 

110 # socket supports setblocking() 

111 setblocking(0)
