36 for websocket and long-polling transports. 

37

39 reconnect to the server after an interruption, or 

40 ``False`` to not reconnect. The default is ``True``. 

41 :param reconnection_attempts: How many reconnection attempts to issue 

42 before giving up, or 0 for infinity attempts. 

43 The default is 0. 

44 :param reconnection_delay: How long to wait in seconds before the first 

45 reconnection attempt. Each successive attempt 

46 doubles this delay. 

47 :param reconnection_delay_max: The maximum delay between reconnection 

48 attempts. 

49 :param randomization_factor: Randomization amount for each delay between 

50 reconnection attempts. The default is 0.5,

54 use. To disable logging set to ``False``. The default is

61 effect on Python 3, where text and binary payloads are 

62 always automatically discovered.

64 packets. Custom json modules must have ``dumps`` and ``loads``

66 versions. 

67

69 

70 :param request_timeout: A timeout in seconds for requests. The default is 

71 5 seconds. 

72 :param ssl_verify: ``True`` to verify SSL certificates, or ``False`` to 

73 skip SSL certificate verification, allowing

75 The default is ``True``. 

76 :param engineio_logger: To enable Engine.IO logging set to ``True`` or pass

557 def _handle_reconnect(self): 

558 self._reconnect_abort.clear() 

559 reconnecting_clients.append(self) 

560 attempt_count = 0 

561 current_delay = self.reconnection_delay 

562 while True: 

563 delay = current_delay 

564 current_delay *= 2 

565 if delay > self.reconnection_delay_max: 

566 delay = self.reconnection_delay_max 

567 delay += self.randomization_factor * (2 * random.random() - 1) 

568 self.logger.info( 

569 'Connection failed, new attempt in {:.02f} seconds'.format( 

570 delay)) 

571 if self._reconnect_abort.wait(delay): 

572 self.logger.info('Reconnect task aborted') 

573 break 

574 attempt_count += 1 

575 try: 

576 self.connect(self.connection_url, 

577 headers=self.connection_headers, 

578 transports=self.connection_transports, 

579 namespaces=self.connection_namespaces, 

580 socketio_path=self.socketio_path) 

581 except (exceptions.ConnectionError, ValueError): 

582 pass 

583 else: 

584 self.logger.info('Reconnection successful') 

585 self._reconnect_task = None 

586 break 

587 if self.reconnection_attempts and \ 

588 attempt_count >= self.reconnection_attempts: 

589 self.logger.info( 

590 'Maximum reconnection attempts reached, giving up') 

591 break 

592 reconnecting_clients.remove(self)

599 def _handle_eio_message(self, data): 

600 """Dispatch Engine.IO messages.""" 

601 if self._binary_packet: 

602 pkt = self._binary_packet 

603 if pkt.add_attachment(data): 

604 self._binary_packet = None 

605 if pkt.packet_type == packet.BINARY_EVENT: 

606 self._handle_event(pkt.namespace, pkt.id, pkt.data) 

607 else: 

608 self._handle_ack(pkt.namespace, pkt.id, pkt.data) 

609 else: 

610 pkt = packet.Packet(encoded_packet=data) 

611 if pkt.packet_type == packet.CONNECT: 

612 self._handle_connect(pkt.namespace) 

613 elif pkt.packet_type == packet.DISCONNECT: 

614 self._handle_disconnect(pkt.namespace) 

615 elif pkt.packet_type == packet.EVENT: 

616 self._handle_event(pkt.namespace, pkt.id, pkt.data) 

617 elif pkt.packet_type == packet.ACK: 

618 self._handle_ack(pkt.namespace, pkt.id, pkt.data) 

619 elif pkt.packet_type == packet.BINARY_EVENT or \ 

620 pkt.packet_type == packet.BINARY_ACK: 

621 self._binary_packet = pkt 

622 elif pkt.packet_type == packet.ERROR: 

623 self._handle_error(pkt.namespace, pkt.data) 

624 else: 

625 raise ValueError('Unknown packet type.')
